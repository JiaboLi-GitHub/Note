# 1 简介

V8 是 Google 开发的一个高性能 JavaScript 引擎，提供快速、可扩展的 JavaScript 执行环境。V8具有以下特点：

- 高性能
- 自动垃圾回收机制
- C++接口
- 多线程支持

# 2 基础知识

## 2.1 Handle

Handle分为两种：

- Local<T>：用于表示本地句柄，指向当前上下文中的对象。
- Persistent<T>：用于表示持久句柄，可以在多个执行上下文中使用。

HandleScope用于管理Handle的生命周期。用于使用Local时，需要声明一个HandleScope的实例。当在HandleScope的作用域内创建Local时，HandleScope会自动管理这些句柄的生命周期，即在HandleScope析构时释放Local。

Persistent的生命周期由开发者负责。

## 2.2 Context

Context用于表示执行JS代码的环境，每个Context都包含一个独立的作用域，拥有自己的全局对象、变量和函数。使用Context::Scope可以指定活动上下文。

## 2.3 Isolate

Isolate代表一个独立的JS执行环境，每个Isolate都有自己的堆内存、全局对象和其他资源，允许在同一个进程中并行允许多个JS上下文。

# 3 使用

## 3.1 基础流程

```c++
#include <v8.h>
#include <iostream>
#include "libplatform/libplatform.h"
#include <assert.h>
#include <windows.h>

using namespace v8;
using namespace std;

int main(int argc, char* argv[]) {
    // 初始化 V8 引擎
    V8::InitializeICUDefaultLocation(argv[0]);
    V8::InitializeExternalStartupData(argv[0]);

    // 创建并初始化 V8 平台
    auto v8_platform = platform::NewDefaultPlatform();
    V8::InitializePlatform(v8_platform.get());
    V8::Initialize();

    // 创建 V8 隔离环境（Isolate）
    Isolate::CreateParams create_params;
    create_params.array_buffer_allocator = ArrayBuffer::Allocator::NewDefaultAllocator();
    Isolate* isolate = Isolate::New(create_params);

    {
        // 在隔离环境中设置作用域
        HandleScope handle_scope(isolate);

        // 创建 JavaScript 执行环境
        Local<Context> context = Context::New(isolate);

        // 设置当前上下文为活动上下文
        Context::Scope context_scope(context);

        std::string js_code = "function add(x, y) { return x + y; } add(1, 2);";

        // 将 C++ 字符串转换为 V8 字符串
        Local<String> code = String::NewFromUtf8(isolate, js_code.c_str(), NewStringType::kNormal, static_cast<int>(js_code.length())).ToLocalChecked();

        // 编译和执行 JavaScript 代码
        Local<Script> script = Script::Compile(isolate->GetCurrentContext(), code).ToLocalChecked();
        Local<Value> result = script->Run(isolate->GetCurrentContext()).ToLocalChecked();

        // 输出 JavaScript 代码的结果
        String::Utf8Value result_str(isolate, result);
        std::cout << "JavaScript Result: " << *result_str << std::endl;
    }

    // 清理 V8 资源
    isolate->Dispose();
    V8::Dispose();
    V8::ShutdownPlatform();

    return 0;
}
```

Script::Run返回JS代码最后执行语句的返回值。

## 3.2 函数模板

主要功能：

1. JS调用C++函数
2. C++调用JS函数

```c++
/*
    Demo_2：将C++函数传入JS中使用
*/

#include <v8.h>
#include <iostream>
#include "libplatform/libplatform.h"
#include <assert.h>
#include <windows.h>

using namespace v8;
using namespace std;

void add(const FunctionCallbackInfo<Value>& args) 
{
    Isolate* isolate = args.GetIsolate();

    // 检查参数数量
    if (args.Length() != 2) 
    {
        isolate->ThrowException(String::NewFromUtf8(isolate, "Two arguments required").ToLocalChecked());
        return;
    }

    // 获取参数并计算和
    if (!args[0]->IsNumber() || !args[1]->IsNumber()) 
    {
        isolate->ThrowException(String::NewFromUtf8(isolate, "Both arguments must be numbers").ToLocalChecked());
        return;
    }

    double sum = args[0]->NumberValue(isolate->GetCurrentContext()).ToChecked() + args[1]->NumberValue(isolate->GetCurrentContext()).ToChecked();

    // 返回结果
    args.GetReturnValue().Set(Number::New(isolate, sum));
}

int main(int argc, char* argv[]) {
    // 初始化 V8 引擎
    V8::InitializeICUDefaultLocation(argv[0]);
    V8::InitializeExternalStartupData(argv[0]);

    // 创建并初始化 V8 平台
    auto v8_platform = platform::NewDefaultPlatform();
    V8::InitializePlatform(v8_platform.get());
    V8::Initialize();

    // 创建 V8 隔离环境（Isolate）
    Isolate::CreateParams create_params;
    create_params.array_buffer_allocator = ArrayBuffer::Allocator::NewDefaultAllocator();
    Isolate* isolate = Isolate::New(create_params);

    {
        // 在隔离环境中设置作用域
        HandleScope handle_scope(isolate);

        // 创建 JavaScript 执行环境
        Local<Context> context = Context::New(isolate);

        // 设置当前上下文为活动上下文
        Context::Scope context_scope(context);

        // 将 C++ 加法函数暴露给 JavaScript
        Local<FunctionTemplate> add_template = FunctionTemplate::New(isolate, add);
        context->Global()->Set(context, String::NewFromUtf8(isolate, "add").ToLocalChecked(), add_template->GetFunction(context).ToLocalChecked()).FromJust();

        // 执行 JavaScript 代码，调用 ADD 函数
        Local<String> source = String::NewFromUtf8(isolate, "add(5, 3);").ToLocalChecked();
        Local<Script> script = Script::Compile(context, source).ToLocalChecked();
        Local<Value> result = script->Run(isolate->GetCurrentContext()).ToLocalChecked();

        // 输出 JavaScript 代码的结果
        String::Utf8Value result_str(isolate, result);
        std::cout << "JavaScript Result: " << *result_str << std::endl;
    }

    // 清理 V8 资源
    isolate->Dispose();
    V8::Dispose();
    V8::ShutdownPlatform();

    return 0;
}
```

```
/*
    Demo_3：C++代码执行JS函数
*/

#include <v8.h>
#include <iostream>
#include "libplatform/libplatform.h"
#include <assert.h>
#include <windows.h>

using namespace v8;
using namespace std;

// C++ 函数，用于调用 JavaScript 函数
void callJsFunction(const FunctionCallbackInfo<Value>& args) {
    Isolate* isolate = args.GetIsolate();

    // 获取 JavaScript 函数
    Local<Function> jsFunction = Local<Function>::Cast(args[0]);

    // 定义参数
    Local<Value> argv[2] = {
        Number::New(isolate, 5),   // 第一个参数
        Number::New(isolate, 3)    // 第二个参数
    };

    // 调用 JavaScript 函数
    Local<Value> result = jsFunction->Call(isolate->GetCurrentContext(), Null(isolate), 2, argv).ToLocalChecked();

    // 输出结果
    double sum = result->NumberValue(isolate->GetCurrentContext()).ToChecked();
    std::cout << "Result from JavaScript add function: " << sum << std::endl;
}

int main(int argc, char* argv[]) {
    // 初始化 V8 引擎
    V8::InitializeICUDefaultLocation(argv[0]);
    V8::InitializeExternalStartupData(argv[0]);

    // 创建并初始化 V8 平台
    auto v8_platform = platform::NewDefaultPlatform();
    V8::InitializePlatform(v8_platform.get());
    V8::Initialize();

    // 创建 V8 隔离环境（Isolate）
    Isolate::CreateParams create_params;
    create_params.array_buffer_allocator = ArrayBuffer::Allocator::NewDefaultAllocator();
    Isolate* isolate = Isolate::New(create_params);

    {
        // 在隔离环境中设置作用域
        HandleScope handle_scope(isolate);

        // 创建 JavaScript 执行环境
        Local<Context> context = Context::New(isolate);

        // 设置当前上下文为活动上下文
        Context::Scope context_scope(context);

        // 定义 JavaScript 函数
        Local<String> jsSource = String::NewFromUtf8(isolate, "function add(x, y) { return x + y; }").ToLocalChecked();
        Local<Script> jsScript = Script::Compile(context, jsSource).ToLocalChecked();
        jsScript->Run(context);

        // 获取 JavaScript 函数
        Local<Value> jsFunctionValue = context->Global()->Get(context, String::NewFromUtf8(isolate, "add").ToLocalChecked()).ToLocalChecked();
        Local<Function> jsFunction = Local<Function>::Cast(jsFunctionValue);

        // 调用 C++ 函数并传入 JavaScript 函数
        Local<FunctionTemplate> tpl = FunctionTemplate::New(isolate, callJsFunction);
        Local<Function> cppFunction = tpl->GetFunction(context).ToLocalChecked();
        Local<Value> argv[1] = { jsFunction };
        cppFunction->Call(context, Null(isolate), 1, argv).ToLocalChecked();
    }

    // 清理 V8 资源
    isolate->Dispose();
    V8::Dispose();
    V8::ShutdownPlatform();

    return 0;
}
```

## 4.2 垃圾回收机制

V8依据代际假说，将堆内存划分为新生代和老生代两个区域。新生代中存放生命周期短的数据，老生代中存放生命周期长的数据。

老生代的内存空间较大，新生代的内存空间较少。

主垃圾回收器负责管理老生代内存空间，副垃圾回收期负责管理新生代内存空间。

副垃圾回收器采用的策略是：将内存空间对半划分为两个区域，一半是From空间，即活跃区域，一半是To空间，即空闲区域。新的数据分配在活跃区域，活跃区域即将满时进行垃圾回收，执行数据晋升策略，将经过两次垃圾回收依然还存活的数据晋升为老生代。之后将其他对象从对象区域拷贝到空闲区域，并交互两个区域的属性。

主垃圾回收期采用的策略是：标记-清除，即遍历堆中的所有对象，未被标记的对象认为是垃圾，进行清除。标记-整理：将所有存活的对象移动到堆的一个端，整理出一块连续的内存区域。
